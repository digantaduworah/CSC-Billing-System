function login(){ let e = document.getElementById("email").value; let p = document.getElementById("pass").value; firebase.auth().signInWithEmailAndPassword(e, p) .then(()=>{ alert("Login Success"); location="app.html"; }) .catch(err=>{ alert(err.message); }); } function signup(){ let e = document.getElementById("email").value; let p = document.getElementById("pass").value; firebase.auth().createUserWithEmailAndPassword(e, p) .then(()=>{ alert("Account Created! Now login"); }) .catch(err=>{ alert(err.message); }); }
